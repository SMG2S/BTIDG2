61void brainAdjMatrixCSR(csr *M_CSR, MatrixDist BlockInfo, Brain *brain, int *neuron_types, BrainMatrixInfo *debugInfo)
62{
63 int my_rank;
64 MPI_Comm_rank(MPI_COMM_WORLD, &my_rank);
65
66 long dim_l = BlockInfo.dim_l;
67 long dim_c = BlockInfo.dim_c;
68 long size = dim_l * dim_c;
69
70 (*M_CSR).dim_l = dim_l;
71 (*M_CSR).dim_c = dim_c;
72
73 long *nb_connections_local_tmp = NULL; //utilisé seulement si debugInfo != NULL
74
75 if (debugInfo != NULL)
76 {
77 (*debugInfo).dim_l = dim_l;
78 (*debugInfo).dim_c = dim_c;
79 (*debugInfo).types = neuron_types;
80 (*debugInfo).nb_connections = (long *)calloc(brain->dimension, sizeof(long));
81 nb_connections_local_tmp = (long *)calloc(brain->dimension, sizeof(long));
82 }
83
84 (*M_CSR).Row = (int *)malloc((dim_l + 1) * sizeof(int));
85 assert((*M_CSR).Row != NULL);
86 (*M_CSR).Row[0] = 0;
87
88 // --- Phase 1 : comptage des connexions par ligne ---
89 long *row_counts = (long *)calloc(dim_l, sizeof(long));
90 assert(row_counts != NULL);
91
92 double t0 = omp_get_wtime();
93
94 #pragma omp parallel
95 {
96 unsigned int seed = 12345u + 17u * omp_get_thread_num();
97
98 #pragma omp for schedule(static)
99 for (long i = 0; i < dim_l; i++)
100 {
101 int ind_part_source = get_brain_part_ind(BlockInfo.startRow + i, brain);
102 int source_type = neuron_types[BlockInfo.startRow + i];
103 long local_count = 0;
104
105 for (long j = 0; j < dim_c; j++)
106 {
107 if ((BlockInfo.startRow + i) == (BlockInfo.startColumn + j))
108 continue; // pas de diagonale
109
110 int ind_part_dest = get_brain_part_ind(BlockInfo.startColumn + j, brain);
111 double p_conn = brain->brainPart[ind_part_source].probaConnection[
112 source_type * brain->nb_part + ind_part_dest];
113
114 double r = (double)rand_r(&seed) / RAND_MAX;
115 if (r < p_conn)
116 {
117 local_count++;
118
119 if (debugInfo != NULL)
120 {
121 #pragma omp atomic
122 nb_connections_local_tmp[BlockInfo.startRow + i]++;
123 }
124 }
125 }
126 row_counts[i] = local_count;
127 }
128 }
129
130 // --- Construction de Row et calcul du nnz total ---
131 long total_nnz = 0;
132 for (long i = 0; i < dim_l; i++)
133 {
134 total_nnz += row_counts[i];
135 (*M_CSR).Row[i + 1] = total_nnz;
136 }
137
138 (*M_CSR).nnz = total_nnz;
139
140 // --- Allocation exacte de Column & Value ---
141 (*M_CSR).Column = (int *)malloc(total_nnz * sizeof(int));
142 (*M_CSR).Value = (int *)malloc(total_nnz * sizeof(int));
143 assert((*M_CSR).Column && (*M_CSR).Value);
144
145 // --- Phase 2 : remplissage effectif des colonnes ---
146 #pragma omp parallel
147 {
148 unsigned int seed = 12345u + 17u * omp_get_thread_num();
149
150 #pragma omp for schedule(static)
151 for (long i = 0; i < dim_l; i++)
152 {
153 int ind_part_source = get_brain_part_ind(BlockInfo.startRow + i, brain);
154 int source_type = neuron_types[BlockInfo.startRow + i];
155 long pos = (*M_CSR).Row[i];
156
157 for (long j = 0; j < dim_c; j++)
158 {
159 if ((BlockInfo.startRow + i) == (BlockInfo.startColumn + j))
160 continue;
161
162 int ind_part_dest = get_brain_part_ind(BlockInfo.startColumn + j, brain);
163 double p_conn = brain->brainPart[ind_part_source].probaConnection[
164 source_type * brain->nb_part + ind_part_dest];
165
166 double r = (double)rand_r(&seed) / RAND_MAX;
167 if (r < p_conn)
168 {
169 assert(pos < (*M_CSR).Row[i + 1]); // sécurité anti-dépassement
170 (*M_CSR).Column[pos++] = j;
171 }
172 }
173
174 for (long k = (*M_CSR).Row[i]; k < (*M_CSR).Row[i + 1]; k++)
175 (*M_CSR).Value[k] = 1;
176 }
177 }
178
179 double t1 = omp_get_wtime();
180
181 if (my_rank == 0)
182 printf("OpenMP section time: %.4f s (%d threads)\n", t1 - t0, omp_get_max_threads());
183
184 free(row_counts);
185
186 // --- Phase MPI Debug ---
187 if (debugInfo != NULL)
188 {
189 (*debugInfo).total_memory_allocated = total_nnz;
190 MPI_Allreduce(&((*M_CSR).nnz), &((*debugInfo).cpt_values), 1, MPI_LONG, MPI_SUM, MPI_COMM_WORLD);
191 MPI_Allreduce(nb_connections_local_tmp, (*debugInfo).nb_connections,
192 brain->dimension, MPI_LONG, MPI_SUM, MPI_COMM_WORLD);
193 free(nb_connections_local_tmp);
194 }
195}

209void brainTransAdjMatrixCSR(csr *M_CSR, MatrixDist BlockInfo, Brain * brain, int * neuron_types, BrainMatrixInfo * debugInfo)
210{
211 long i,j,cpt_values,size = BlockInfo.dim_l * BlockInfo.dim_c;
212 long * nb_connections_local_tmp; //utilisé seulement si debugInfo != NULL
213 int ind_part_source,ind_part_dest,source_type; double proba_connection,proba_no_connection,random;
214 (*M_CSR).dim_l = BlockInfo.dim_l; (*M_CSR).dim_c = BlockInfo.dim_c;
215 if (debugInfo != NULL)
216 {
217 (*debugInfo).dim_l = BlockInfo.dim_l; (*debugInfo).dim_c = BlockInfo.dim_c;
218 (*debugInfo).types = neuron_types;
219 //Attention : le malloc de (*debugInfo).nb_connections n'est pas free dans la fonction.
220 (*debugInfo).nb_connections = (long *)malloc((*brain).dimension * sizeof(long)); //contiendra le nombre de connexions faites au global (tout processus confondu)
221 nb_connections_local_tmp = (long *)malloc((*brain).dimension * sizeof(long)); //contiendra le nombre de connexions faites en local (décalage en ligne prit en compte)
222 for (i=0;i<(*brain).dimension;i++)
223 {
224 nb_connections_local_tmp[i] = 0;
225 (*debugInfo).nb_connections[i] = 0;
226 }
227 }
228
229 //allocations mémoires
230 (*M_CSR).Row = (int *)malloc(((*M_CSR).dim_l+1) * sizeof(int));
231 //La mémoire allouée pour Column est à la base de 1/10 de la taille de la matrice stockée "normalement". Au besoin, on réalloue de la mémoire dans le code.
232 long basic_size = (long) size/10;
233 long total_memory_allocated = basic_size; //nombre total de cases mémoires allouées pour 1 vecteur
234 (*M_CSR).Column = (int *)malloc(total_memory_allocated * sizeof(int));
235
236 (*M_CSR).Row[0] = 0;
237 cpt_values=0;
238 for (i=0;i<BlockInfo.dim_l;i++) //parcours des lignes
239 {
240 //récupération de l'indice de la partie (du cerveau) destination
241 ind_part_dest = get_brain_part_ind(BlockInfo.startRow+i, brain);
242 for (j=0;j<BlockInfo.dim_c;j++) //parcours des colonnes
243 {
244 //récupération de l'indice de la partie source
245 ind_part_source = get_brain_part_ind(BlockInfo.startColumn+j, brain);
246 //récupération du type de neurone
247 source_type = neuron_types[BlockInfo.startColumn+j];
248 //récupération de la probabilité de connexion source -> destination avec le type de neurone donné
249 proba_connection = (*brain).brainPart[ind_part_source].probaConnection[source_type*(*brain).nb_part + ind_part_dest];
250 proba_no_connection = 1 - proba_connection;
251 random = rand_0_1();
252 //décision aléatoire, en prenant en compte l'abscence de connexion sur la diagonale de façon brute
253 if ( (BlockInfo.startRow+i)!=(BlockInfo.startColumn+j) && random > proba_no_connection) //si on est dans la proba de connexion et qu'on est pas dans la diagonale, alors on place un 1
254 {
255 if (cpt_values >= total_memory_allocated)
256 {
257 total_memory_allocated *= 2;
258 (*M_CSR).Column = (int *) realloc((*M_CSR).Column, total_memory_allocated * sizeof(int));
259 assert((*M_CSR).Column != NULL);
260 }
261 (*M_CSR).Column[cpt_values] = j;
262 if (debugInfo != NULL)
263 {
264 nb_connections_local_tmp[j + BlockInfo.startColumn] = nb_connections_local_tmp[j + BlockInfo.startColumn] + 1;
265 }
266 cpt_values++;
267 }
268 }
269 (*M_CSR).Row[i+1] = cpt_values;
270 }
271 //remplissage du vecteur Value (avec précisement le nombre de 1 nécéssaire)
272 (*M_CSR).Value = (int *)malloc(cpt_values * sizeof(int));
273 (*M_CSR).nnz = cpt_values;
274 for (i=0; i<cpt_values;i++) {(*M_CSR).Value[i] = 1;}
275
276 //remplissage de la structure de débuggage
277 if (debugInfo != NULL)
278 {
279 (*debugInfo).total_memory_allocated = total_memory_allocated;
280 MPI_Allreduce(&((*M_CSR).nnz), &((*debugInfo).cpt_values), 1, MPI_LONG, MPI_SUM, MPI_COMM_WORLD); //somme MPI_SUM des nombres de valeurs dans les sous matrices dans cpt_values, nombre de valeurs non nulles global
281
282 /* MatrixDebugInfo.nb_connections contient actuellement (dans chaque processus) le nombre de connexions faites LOCALEMENT par tout les neurones par colonne. */
283 MPI_Allreduce(nb_connections_local_tmp, (*debugInfo).nb_connections, (*brain).dimension, MPI_LONG, MPI_SUM, MPI_COMM_WORLD); //somme MPI_SUM de tout les nb_non_zeros_local dans (*debugInfo).nb_connections
284 free(nb_connections_local_tmp);
285 /* MatrixDebugInfo.nb_connections contient maintenant (dans tout les processus) le nombre GLOBAL de connexions faites pour chaque neurone. */
286 }
287}
