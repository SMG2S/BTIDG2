69int get_brain_part_ind(long long ind, Brain * brain)
70{
71 /*
72 Renvoie l'indice de la partie du cerveau dans laquelle le neurone "ind" se situe
73 Brain est supposé être un cerveau bien formé et ind est supposé être entre 0 et brain.dimension
74 */
75 long long * parts_cerv = (*brain).parties_cerveau;
76 if (ind >= parts_cerv[(*brain).nb_part - 1])
77 {
78 return (*brain).nb_part - 1;
79 }
80 int i=0;
81 while (ind >= parts_cerv[i+1])
82 {
83 i++;
84 }
85 return i;
86}

95int get_nb_neuron_brain_part(Brain * brain, int part)
96{
97 long long n = (*brain).dimension;
98 long long ind_depart = (*brain).parties_cerveau[part];
99 if (part+1 == (*brain).nb_part)
100 {
101 return n - ind_depart;
102 }
103 else
104 {
105 long long ind_fin = (*brain).parties_cerveau[part+1];
106 return ind_fin - ind_depart;
107 }
108}

118double get_mean_connect_percentage_for_part(Brain * brain, int part, int type)
119{
120 long long n,i;
121 int nb_part;
122 double * probCo = (*brain).brainPart[part].probaConnection;
123 n = (*brain).dimension;
124 nb_part = (*brain).nb_part;
125
126 double sum_proba = 0;
127 for (i=0;i<nb_part;i++)
128 {
129 sum_proba += (double) get_nb_neuron_brain_part(brain,i) * (*brain).brainPart[part].probaConnection[type*nb_part + i];

167void generate_neuron_types(Brain * brain, int ind_start_neuron, int nb_neuron, int * types)
168{
169 /*
170 Decides the types of neurons number "ind_start_neuron" to "ind_start_neuron + nb_neuron" in the brain "Brain", and writes them in "types"
171 A malloc of size nb_neuron * sizeof(int) must have been made beforehand for the "types" pointer.
