92 */

95 if (indl >= (*M_CSR).dim_l || indc >= (*M_CSR).dim_c)

145MatrixDist initMatrixDist(int rank, int nb_blocks_row, int nb_blocks_column, long n)
146{
147 struct MatrixDist MBlock;
148 MBlock.indl = rank / nb_blocks_column;
149 MBlock.indc = rank % nb_blocks_column;
150 MBlock.dim_l = n/nb_blocks_row;
151 MBlock.dim_c = n/nb_blocks_column;
152 MBlock.startRow = MBlock.indl*MBlock.dim_l;
153 MBlock.endRow = (MBlock.indl+1)*MBlock.dim_l-1;
154 MBlock.startColumn = MBlock.indc*MBlock.dim_c;
155 MBlock.endColumn = (MBlock.indc+1)*MBlock.dim_c-1;
156 return MBlock;
157}
